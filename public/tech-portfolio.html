1️⃣ 動態表單驗證器 - Validator Factory Pattern

/**
 * 建立數量驗證器 - 展示高階函數與閉包應用
 * 支援動態角色權限檢查
 */
export const createQuantityValidator = (
  contextGetter: () => CheckQuantityContext
) => {
  return (rule, value, callback) => {
    const { quantity, role } = contextGetter();
    
    // 角色權限檢查
    if (role !== 'SUPPLIER') {
      return callback();
    }
    
    // 必填驗證
    if (!quantity || quantity === 0) {
      return callback(new Error('數量為必填'));
    }
    
    // 正整數驗證
    if (!/^\d+$/.test(String(quantity))) {
      return callback(new Error('數量必須為正整數'));
    }
    
    callback();
  };
};

3️⃣ 權限控制邏輯 - 函數式編程

/**
 * 動態解析欄位的 ReadMode 狀態
 * 支援函數式與靜態兩種配置方式
 */
const resolveReadMode = (
  field: SampleInfomationInputFieldConfig
) => {
  if (typeof field.isReadMode === 'function') {
    return field.isReadMode({
      model: props.modelValue,
      role: getAuthStore.mainRole,
      status: getSamplingStore.status
    });
  }
  return !!field.isReadMode;
};

// 使用範例：動態權限配置
const fieldConfig = {
  isReadMode: ({ role, status }) => 
    role === 'CE' && status === 'APPROVED'
};
